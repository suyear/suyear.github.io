pagination分页的html以及js代码如下

分页功能依赖于jquery.paginate.css，jquery.paginate.js以及template.js文件


	//分页HTML
	<div id="pagination1">
	</div>
	

	//分页JS
	<script type="text/javascript">
	  	var currentPage = 1;
	  	requestData(currentPage);
	  	//请求数据
	  	function requestData(pageNumber){
		   $.ajax({
			type: "get",
			url: "img/test1.json",
			data: {"pageNumber":pageNumber},
			dataType: "json",
			success: function(data){
				show(data)
			},
			error: function(){
				alert("error");
			}
		   });
		}
		//展示数据
		function show(data){
			//渲染数据
			var dataContainer = $(".data-container1");
			dataContainer.empty();
			$.each(data.persons,function(i,item){
			  var html =  template('datatemplate', item);				        	
			  dataContainer.append(html);	
			});
			//分页
			var totalPage = data.totalPage;
			$('#pagination1').paginate({
				count: totalPage,
				start: currentPage,
				onChange: function(page){
				  requestData(currentPage);
				  currentPage = page
				}
			})
		}
	</script>
	

	//数据模板
	<script type="text/html" id="datatemplate">
	  <tr>
	     <td class="col-xs">
	      <label class="checkbox">
	     	<input class="input-checkbox" value="2" type="checkbox">
	     	<span class="style-checkbox"></span>
	      </label>
	     </td>
	     <td class="col-md">
	     	<span>
	     	  <a href="javascript:;">{{name}}</a>
	     	</span>
	     </td>
	     <td class="col-sm">
	     	<span>{{age}}</span>
	     </td>
	     <td class="col-lg">
	     	<span>{{number}}</span>
	     </td>
	     <td class="col-lg">
	     	<span>{{tel}}</span>
	     </td>
	     <td class="col-xlg">
	     	<span>{{address}}</span>
	     </td>
	  </tr>
	</script>
